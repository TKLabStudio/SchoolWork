package ch2;
//阿就我寫的啊
class Product{
  public static int calculate (int n)
  {
    if(n>1)
      return n*calculate(n-1);
    else
      return n;
  }
}
public class ch10_p7 { 
  
  public static void main(String[] args) {
    Product p = new Product();
    System.out.print(p.calculate(10));

  }

}
